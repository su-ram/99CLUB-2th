def solution(citations):
    left = 0
    right = max(citations)
    answer = 0    
    while(right >= left):
        h = (left + right) // 2
        biggerCnt = 0
        smallerCnt = 0
        for c in citations:
            if c >= h: biggerCnt += 1
            if c <= h: smallerCnt += 1 
                
        if biggerCnt >= h:
            answer = h
            left =  h + 1 
        else:
            right = h - 1 
    return answer
